3. Layout mit grid-template-areas

.areas {
    display: grid;
    grid-template-areas:
        "header header header"
        "nav content content"
        "footer footer footer";
    grid-template-columns: 300px 800px 200px;
    grid-template-rows: 250px 400px 250px;
}
Header 
.header {
    grid-area: header;
    background: darkslateblue;
}
Nav 
.nav {
    grid-area: nav;
    background: darkgoldenrod;
}
Content 
.content {
    grid-area: content;
    background: darkolivegreen;
}
Footer 
.footer {
    background-color: darkorange;
    grid-area: footer;
}
